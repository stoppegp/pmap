 99 /**
100  * Zeigt eine Gruppe im Info-Panel an

103  * @param {function()} [callback] - Callback-Funktion, wird am Ende der Animation ausgeführt
104  */
105 function showGruppe( key, key2 = undefined, callback = null ) {
106     // Wenn passendes Panel bereits offen und richtiges Treffen ausgewählt: Nichts tun, außer Callback
107     if (infoOpen === true && infoData !== undefined && infoData.key !== undefined && infoData.key === key && infoData.key2 === key2) { if (callback !== null) callback(); return; }
108 
109     // Wenn passendes Panel bereits offen und falsches Treffen ausgewählt: Richtiges Treffen auswählen, Shariff und Callback
110     if (infoOpen === true && infoData !== undefined && key === infoData.key && key2 !== infoData.key2) {
111         $(".treffenlink").removeClass("active");
112         if (key2 !== undefined) {
113             $("#treffen-" + key + "-" + key2).addClass("active");            
114             initShariff(key, genLink([key, key2]), genTitle(key, key2));
115         } else {
116             initShariff(key, genLink([key]), genTitle(key));
117         }
118         infoData = {"key": key, "key2": key2};
119         
120         if (callback !== null) callback();
121         return;
122     }
123 
124     // Ansonsten
125 
126     // Aktiv-Klassen entfernen, Aktiv-Klasse zur Gruppe hinzufügen
127     $(".activegroup").removeClass("active");
128     $(".treffenlink").removeClass("active");
129     $("#activegroup-" + key).addClass("active");
130 
131     var gruppe = pdata[key];
132 
133     // HTML erstellen
134     var html_gruppe = "<h2>" + gruppe.name + "</h2>";
135     if (gruppe.img !== undefined) {
136             html_gruppe += "<img class=\"kreislogo\" src=\"" + basePath + "/data/img/" + gruppe.img + "\" alt=\"" + gruppe.name + "\">";
137     }
138     if (gruppe.text) html_gruppe += "<p class=\"text\">" + gruppe.text + "</p>";
139     if (gruppe.homepage) html_gruppe += "<p><strong>Homepage:</strong><br><a href=\"" + gruppe.homepage + "\">" + gruppe.homepage + "</a></p>";
140     if (gruppe.email) html_gruppe += "<p><strong>E-Mail:</strong><br><a href=\"mailto:" + gruppe.email + "\">" + gruppe.email + "</a></p>";
141     html_gruppe += '<p><div class="shariff" id="shariff' + key + '"></div></p>';
142 
143     // Übersicht der Treffen
144     if (pdata[key].treffen !== undefined) {
145         html_gruppe += "<h3>Treffen</h3>";
146         $.each( pdata[key].treffen, function( key2a, val ) {
147             var exclass = "";

151             if (val.ort) html_gruppe += "<br>" +  val.ort;
152             if (events !== undefined && events[key] !== undefined && events[key].treffen !== undefined && events[key].treffen[key2a] !== undefined) {
153                     var datum = moment.unix(events[key].treffen[key2a]);
154                     html_gruppe += "<br><small><em>Nächster Termin: " + datum.format("ddd, DD.MM.YYYY, HH:mm") + " Uhr</em></small>";
155             }
156             html_gruppe += "</a></p>";

158     }
159

161     if (events && events[key] && events[key].ical) {
162         html_gruppe += "<h3>Kommende Termine</h3>";
163         $.each( events[key].ical, function( key, val ) {
164             datum = moment.unix(val.start);
165             datume = moment.unix(val.end);
166             if (datume.isBefore(moment(), 'hour')) return true;
167             html_gruppe += "<p><small>" + datum.format("ddd, DD.MM.YYYY, HH:mm") + " Uhr</small><br>";

262 /**
263  * PLZ-Suche starten
264  * @param {number} plz - Postleitzahl
265  */
266 function startPLZ( plz ) {
267     $( "#plz" ).val(plz);   // Suchfeld füllen
268 
269     // Wenn PLZ-Daten noch nicht geladen wurden, erst laden und Funktion erneut aufrufen
270     if (!plzdata) {
271         $.getJSON( "/res/plz.json", function( data ) {
272             plzdata = data;
273             startPLZ( plz );
274         });
275     } else if (plzdata[plz]) {
276         var dat = plzdata[plz]; 
277         plzmarker.setLatLng([dat.lat, dat.lon]).addTo(map);
278         var key = "0" + dat.ags.toString().substr(0,4);
279         plzmarker.plz = plz;
280         if (kreiskeys[key] !== undefined) {
281             // Gruppe zur Suche gefunden
282             plzpopup.setContent("<strong>" + dat.name + "</strong>");
283             showGruppe(kreiskeys[key]);
284             plzmarker.openPopup();
285         } else {
286             // Keine Gruppe gefunden
287             if (!plzpopup.isOpen()) closeAllPopups();
288             plzpopup.setContent("<strong>" + dat.name + "</strong>");
289             showEmpty( dat.name );
290             plzmarker.openPopup();
291         }
292     }

308 /**
309  * Gruppe starten
310  * @param {number} key - key der Gruppe
311  * @param {(boolean|Object.<L.LatLng, number>)} [zoom=false] - true für Zoom, L.LatLng&Zoom für setView
312  * @param {function()} [callback] - Callback-Funktion, wird nach der Animation ausgeführt
313  */
314 function startGruppe( key , zoom = false, callback = null) {
315     if (pdata[key].popup !== undefined) {
316         // Wenn richtiges Popup nicht geöffnet, werden vorher alle anderen Popups geschlossen
317         if (!pdata[key].popup.isOpen()) { closeAllPopups(); pdata[key].layer.openPopup(); }
318     } else {
319         closeAllPopups();
320         // Wenn kein Popup vorhanden, wird die Karte bei kleinen Bildschirmen manuell verschoben
321         if (infoOpen === false && screenState === "smallportrait") map.panBy([0, infoSize], {"duration": 0.5});
322         if (infoOpen === false && screenState === "smalllandscape") map.panBy([infoSize, 0], {"duration": 0.5});
323     }
324     showGruppe( key, undefined, callback );
325 
326     // Karte anpassen
327     if (zoom === true && pdata[key].layer !== undefined) {
328         map.fitBounds(pdata[key].layer.getBounds(), {"paddingTopLeft": pTLi, "paddingBottomRight" : pBRi});
329     } else if (zoom.center !== undefined && zoom.zoom !== undefined) {
330         map.setView(zoom.center, zoom.zoom);
331     }	
332 }

361 /**
362  * Neuen History-Eintrag hinzufügen und Titel setzen
363  * @param {(Number|string)} key - key der Gruppe
364  * @param {Number} key2 - key des Treffens
365  * @param {boolean} replace - Wenn true, wird der aktuelle History-Eintrag ersetzt
366  */
367 function setHash(key = undefined, key2 = undefined, replace = false) {
368     if (key !== undefined && key2 !== undefined) var newhash = genLink([key, key2]); else newhash = genLink([key]);
369     if (newhash.substr(1) !== window.location.pathname.substr(1) || replace === true) {
370         setTimeout(function () {
371             var state = {"key": key, "key2": key2, "mapCenter": map.getCenter(), "mapZoom": map.getZoom()};
372             if (replace === true) history.replaceState(state, null, newhash); else history.pushState(state, null, newhash);
373             setTitle(key, key2);
374         }, 100);		
375     }
376 }

378 /**
379  * Link generieren
380  * @param {Array.number,Object.<number, number>),(Object.<string, number>)} keys - keys
381   * @returns {String}  
382  */
383 function genLink(keys) {
384     if (keys === undefined) return baseURL + "/";
385     if (keys.length === 2 && keys[0] !== undefined && keys[1] !== undefined) {
386         if (keys[0] === "plz" && $.isNumeric(keys[1])) return baseURL + "/plz/" + keys[1]; else if (keys[0] === "plz") return baseURL + "/";
387         var part1, part2;
388         if (pdata[keys[0]].slug !== undefined) part1 = pdata[keys[0]].slug; else part1 = keys[0];		
389         if (pdata[keys[0]] !== undefined && pdata[keys[0]].treffen !== undefined && pdata[keys[0]].treffen[keys[1]].slug !== undefined) part2 = pdata[keys[0]].treffen[keys[1]].slug; else part2 = keys[1];	
390         return baseURL + "/" + part1 + "/" + part2;	
391     } else if (keys.length === 1 && keys[0] !== undefined) {
392         if (pdata[keys[0]].slug !== undefined) return baseURL + "/" + pdata[keys[0]].slug; else return baseURL + "/" + keys[0];		
393     }
394 
395     return baseURL + "/" + keys.join("/");
396 }

463         // smallportrait

465         screenState = "smalllandscape";

476             // Bei geöffnetem Info-Panel: Common-Panel ausblenden & Leaflet-Attribution verschieben
477             $("#common").css("width", "0px");
478             $(".leaflet-left").css("left", 0);
479         } else {
480             // Bei geschlossenem Info-Panel: Common-Panel einblenden & Leaflet-Attribution verschieben
481             $("#common").css("width", commonSize + "px");
482             $(".leaflet-left").css("left", commonSize + "px");
483             $("#info").hide();
484         }
485 
486         pTLc = [commonSize, 0];
487         pBRc = [0, 0];
488         pTLi = [0, 0];
489         pBRi = [infoSize, 0];
490     } else {
491         // big
492         // Common-Panel links, Info-Panel rechts.
493         screenState = "big";
494         $("body").removeClass("smallportrait");
495         $("body").removeClass("smalllandscape");
496         commonSize = w/5;
497         if (commonSize < 250) commonSize = 250;
498         infoSize = w/4;
499         if (infoSize < 250) infoSize = 250;

502         $("#common").css({"width": commonSize + "px", "height": ""});
503         $("#commonc").css({"width": commonSize + "px", "height": ""});
504 
505         if (infoOpen === false) $("#info").hide();
506 
507         $(".leaflet-left").css("left", commonSize + "px");
508 
509         pTLc = [$("#common").width(), 0];
510         pBRc = [0, 0];
511         pTLi = [$("#common").width(), 0];
512         pBRi = [infoSize, 0];
513     }
514 }
515 /**
516  * Funktion wird ausgeführt, wenn ein Popup geöffnet wird.
517  * popupOpen wird auf false gesetzt.
518  * Das Popup wird mit den aktuellen Padding-Werten aktualisiert.
519  * @param {L.Popup} popup - Das Popup-Objekt
520  */
521 function onPopupAdd( popup ) {
522     popupOpen = true;
523     popup.options.autoPanPaddingTopLeft = pTLi;
524     popup.options.autoPanPaddingBottomRight = pBRi;
525     popup.update();
526 }

528 /**
529  * Funktion wird ausgeführt, wenn ein Popup geschlossen wird.
530  * Schließt das Info-Panel. Kurze Wartezeit um abzufragen, ob vorher
531  * ein anderes Popup geöffnet wurde. popupOpen wird auf false gesetzt.
532  * popupToClose wird abgefragt um festzustellen, ob das Popup per Funktionsaufruf
533  * geschlossen wurde. In diesem Fall wird keine Aktion ausgeführt.
534  */
535 function onPopupRemove() {
536     if (popupToClose === true) {
537             popupToClose = false;
538             return;
539     }
540     popupOpen = false; 
541     setTimeout(function() {
542             if (popupOpen === false) {
543                     closeInfo();
544                     setHash();
545             }
546     }, 50);
547 }
548 
549 /**
550  * Schließt alle Popups, sofern welche geöffnet sind.
551  */
552 function closeAllPopups() {
553     if (popupOpen === false) return;
554     popupToClose = true;
555     map.closePopup();
556 }
